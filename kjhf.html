def greet(name):
    """A simple greeting function"""
    return f"Hello, {name}!"

# Example usage
user_name = "World"
message = greet(user_name)
print(message)

# More examples
numbers = [1, 2, 3, 4, 5]
squared = [x**2 for x in numbers]
print(f"Original: {numbers}")
print(f"Squared: {squared}")
